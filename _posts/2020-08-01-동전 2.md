* 재귀를 시도해봤지만 시간초과가 나왔다

```python

n, K = map(int, input().split())

coins = set()
for i in range(n):
    coins.add(int(input()))

coins = list(coins)

coins.sort()

length = len(coins)

arr = []
arr.append(0)

for i in range(1, K+1):
    min = 10000000
    for j in range(length):
        if (i - coins[j]) >= 0:
            if min > arr[i-coins[j]]:
                min = arr[i-coins[j]]
        else:
            break
    arr.append(min+1)

if arr[K] == 10000001:
    print(-1)
else:
    print(arr[K])
    
```
