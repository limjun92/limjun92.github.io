import java.util.Scanner;

public class Solution {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int T = sc.nextInt();
		for (int tc = 1; tc <= T; tc++) {

			int N = sc.nextInt();
			int M = sc.nextInt();
			int K = sc.nextInt();

			int[] second = new int[11112];

			for (int i = 0; i < N; i++) {
				int index = sc.nextInt();
				second[index]++;
			}

			System.out.print("#" + tc + " ");

			int time = 0;
			if (second[time] != 0)
				System.out.print("Impossible");
			else {
				int count = 0;
				int n_count = 0;
				while (true) {
					time++;
					if (time % M == 0)
						count += K;
					if (second[time] != 0) {
						count -= second[time];
						n_count += second[time];
					}
					if (count < 0) {
						System.out.print("Impossible");
						break;
					}
					if (n_count == N) {
						System.out.print("Possible");
						break;
					}
				}
			}
			System.out.println();
		}
	}
}
