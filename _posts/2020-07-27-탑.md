* class 배열 구현
* 스택을 사용하지 않고 구현

class top:
    def __init__(self, index, value):
        self.index = index
        self.value = value

N = int(input())

top_list = []

arr = list(map(int, input().split(" ")))

top_list.append(top(1,arr[0]))
print(0)

for i in range(2,N+1):
    n = arr[i-1]
    if n > top_list[0].value:
        top_list = []
        top_list.append(top(i,n))
        print(0)
        continue

        continue
