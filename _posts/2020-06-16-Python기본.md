* 직접 key값을 지정해서 정렬

```python
pairs = [
    ('time', 8),
    ('the', 15),
    ('turbo', 1),
]
 
def get_freq(pair):
    return pair[1]

def sort_by_frequency(pairs):
    return sorted(pairs,key = get_freq)
    # get_freq를 key롤 지정해서 정렬한다
    
print(sort_by_frequency(pairs))
# [('turbo', 1), ('time', 8), ('the', 15)]
```

# 집합

* 집합은 중복을 허용하지 않는다.

```
a = set()
b = set([1,2,3,3,3,3,4,5])
print(b) # {1,2,3,4,5}
print(type(b)) # <class 'set'>

s1 = set([1,2,3,4,5])
s2 = set([4,5,6,7,8])

# * 교집합
print(s1 & s2) # {4,5}
print(s1.intersection(s2)) # {4,5}

# * 합집합
print(s1 | s2) # {1,2,3,4,5,6,7,8}
print(s1.union(s2)) # {1,2,3,4,5,6,7,8}

# * 차집합
print(s1 - s2) # {1,2,3}
print(s1.difference(s2)) # {1,2,3}

a = set([1,2,3,4,5])
a.add(7)
print(a) # {1, 2, 3, 4, 5, 7}

a.update([10,11,12])
print(a) # {1, 2, 3, 4, 5, 7, 10, 11, 12}

a.remove(5)
print(a) # {1, 2, 3, 4, 7, 10, 11, 12}
```
